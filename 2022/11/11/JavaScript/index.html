流程控制

流程控制主要有三种结构，分别是顺序结构、分支结构和循环结构

if分支语句

if(条件语句) {
    执行语句/*条件语句成立时 执行*/
}
else {
    执行语句/*条件语句不成立时 执行*/
}

三元表达式

条件表达式？表达式1：表达式2
如果条件为真，则返回表达式1，反之则返回表达式2
1>2?1:2 ，返回2

switch语句

语法结构


/*利用表达式的值和case后面的选项进行匹配，全等 
则执行该case*/
switch(表达式) {
    case value1: 
         执行语句1;
         break;
    case value1: 
        执行语句1;
        break;
    case value1: 
        执行语句1;
        break;
    ...

    default:
        执行最后的语句；
}

注意：只有当所有case都匹配不上，才会执行default

switch(1) {
    default://只有匹配不上才会执行
          console.log(4);
    case 1: 
        console.log(1);//输出1
         break;
    case 2: 
        console.log(2);
        break;
}

数组

数组是指一组数据的集合，其中每个数据被称为元素，在数组中可以存放任意类型的元素。

var arr = new Array();//创建一个空的数组
//数组中的元素可以是不同的数据类型
var arr = [];//声明一个空数组
var arr =  [1, 2.3, 'hello' ,true]; //利用数组字面量创建数组
console.log(arr);//访问整个数组数组
console.log(arr[0]);//访问数组中的第一个元素
arr.length;//获取数值长度
//新增数组元素有以下两种方法
//1.修改数组长度 
arr.length += 1; //扩大数组的长度 数组的大小可以任意修改 
console.log(arr[arr.length-1]);//新增的元素为空
//2.修改索引号 追加元素
arr[arr.length] = 1;//原来没有的索引号会追加元素 此时length会自动直接增加

函数

函数就是封装了一段可被重复调用执行的代码块

function getSum(num1,num2) { //function 声明函数的关键字 getSum是函数名 num1和num2是函数的形参
    var = sum;
    for (var i = num1 ; i <= num2 ;i++){
        sum +=i;
    }
    return sum;
    /*函数的返回值 返回给函数的调用者 
    return会终止函数 
    rerun只能返回一个值
    return 1,2; return会返回最后一个值2
    如果函数没有返回值，则会返回undefined
    */
}
var get = function () {//另外一种函数声明的方式
    console.log('13');
}
getSum(1,100);//调用函数 1和100是函数的实参
getSum(1,100,123)//当函数个数实参大于形参个数时，多余的实参不会传给形参 123没用
getSum(0);//当形参 个数大于实参个数时，形参相当于一个没有赋值的变量
